# Test cases for MyList class

>>> MyList = __import__('1-my_list').MyList

# Test empty list
>>> my_list = MyList()
>>> my_list.print_sorted()
[]

# Test list with one element
>>> my_list = MyList([1])
>>> my_list.print_sorted()
[1]

# Test list with multiple elements
>>> my_list = MyList([1, 4, 2, 3, 5])
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]

# Test that original list is not modified
>>> my_list
[1, 4, 2, 3, 5]

# Test list with negative numbers
>>> my_list = MyList([1, -4, 2, -3, 5])
>>> my_list.print_sorted()
[-4, -3, 1, 2, 5]

# Test list with duplicate numbers
>>> my_list = MyList([1, 2, 1, 2, 3])
>>> my_list.print_sorted()
[1, 1, 2, 2, 3]

# Test that MyList inherits from list
>>> issubclass(MyList, list)
True

# Test that print_sorted returns None
>>> my_list = MyList([3, 2, 1])
>>> result = my_list.print_sorted()
[1, 2, 3]
>>> print(result)
None
